```python
from dataclasses import dataclass, field
from datetime import datetime
from typing import Literal

@dataclass
class Relation:
    """Topic 之间的关系"""
    from_topic: str
    to_topic: str
    relation_type: Literal["associated", "parent_child", "decomposed_from", "related_to"]
    created_at: datetime = field(default_factory=datetime.now)

@dataclass
class Topic:
    """
    统一的 Topic 数据模型
    
    兼容现有字段 + 新增图结构字段
    Schema Version: 2.0
    """
    # 基础字段（兼容现有）
    name: str
    known: bool = False
    depth: float = 0.0
    summary: str = ""
    sources: list[str] = field(default_factory=list)
    status: Literal["unexplored", "partial", "complete"] = "unexplored"
    
    # 元认知字段（兼容现有）
    explore_count: int = 0
    marginal_returns: list[float] = field(default_factory=list)
    last_quality: float = 0.0
    
    # 图结构字段（新增）
    parents: list[str] = field(default_factory=list)
    children: list[str] = field(default_factory=list)
    relations: list[Relation] = field(default_factory=list)
    explored_by: list[str] = field(default_factory=list)
    fully_explored: bool = False
    
    # 时间戳
    created_at: datetime = field(default_factory=datetime.now)
    explored_at: datetime | None = None
    fully_explored_at: datetime | None = None
    
    # Schema 版本
    schema_version: str = "2.0"
    
    @property
    def degree(self) -> int:
        """节点的度（入度 + 出度）"""
        return len(self.parents) + len(self.children)
    
    def add_parent(self, parent: str) -> None:
        """添加父节点（去重）"""
        if parent not in self.parents:
            self.parents.append(parent)
    
    def add_child(self, child: str) -> None:
        """添加子节点（去重）"""
        if child not in self.children:
            self.children.append(child)
    
    def mark_explored(self, by: str | None = None) -> None:
        """标记为已探索"""
        self.known = True
        self.explored = True
        self.status = "partial"
        self.explored_at = datetime.now()
        if by and by not in self.explored_by:
            self.explored_by.append(by)
    
    def mark_fully_explored(self) -> None:
        """标记为充分探索"""
        self.fully_explored = True
        self.status = "complete"
        self.fully_explored_at = datetime.now()
    
    def to_dict(self) -> dict:
        """序列化为字典（用于 JSON 存储）"""
        return {
            "name": self.name,
            "known": self.known,
            "depth": self.depth,
            "summary": self.summary,
            "sources": self.sources,
            "status": self.status,
            "explore_count": self.explore_count,
            "marginal_returns": self.marginal_returns,
            "last_quality": self.last_quality,
            "parents": self.parents,
            "children": self.children,
            "relations": [
                {
                    "from": r.from_topic,
                    "to": r.to_topic,
                    "type": r.relation_type,
                    "created_at": r.created_at.isoformat(),
                }
                for r in self.relations
            ],
            "explored_by": self.explored_by,
            "fully_explored": self.fully_explored,
            "created_at": self.created_at.isoformat(),
            "explored_at": self.explored_at.isoformat() if self.explored_at else None,
            "fully_explored_at": self.fully_explored_at.isoformat() if self.fully_explored_at else None,
            "schema_version": self.schema_version,
        }
    
    @classmethod
    def from_dict(cls, data: dict) -> "Topic":
        """从字典反序列化"""
        relations = [
            Relation(
                from_topic=r["from"],
                to_topic=r["to"],
                relation_type=r["type"],
                created_at=datetime.fromisoformat(r["created_at"]),
            )
            for r in data.get("relations", [])
        ]
        
        return cls(
            name=data["name"],
            known=data.get("known", False),
            depth=data.get("depth", 0.0),
            summary=data.get("summary", ""),
            sources=data.get("sources", []),
            status=data.get("status", "unexplored"),
            explore_count=data.get("explore_count", 0),
            marginal_returns=data.get("marginal_returns", []),
            last_quality=data.get("last_quality", 0.0),
            parents=data.get("parents", []),
            children=data.get("children", []),
            relations=relations,
            explored_by=data.get("explored_by", []),
            fully_explored=data.get("fully_explored", False),
            created_at=datetime.fromisoformat(data.get("created_at", datetime.now().isoformat())),
            explored_at=datetime.fromisoformat(data["explored_at"]) if data.get("explored_at") else None,
            fully_explored_at=datetime.fromisoformat(data["fully_explored_at"]) if data.get("fully_explored_at") else None,
            schema_version=data.get("schema_version", "2.0"),
        )
```

```python
"""
Schema 迁移工具
支持从 v1.0 迁移到 v2.0
"""
from typing import Any

def migrate_state_v1_to_v2(v1_state: dict) -> dict:
    """
    迁移 v1 state.json 到 v2 格式
    
    Changes:
    - 添加 schema_version = "2.0"
    - topics: 添加 parents[], relations[], explored_by[], fully_explored
    - 将单 parent 转换为 parents[]
    """
    v2_state = v1_state.copy()
    v2_state["schema_version"] = "2.0"
    
    # 迁移 topics
    if "knowledge" in v2_state and "topics" in v2_state["knowledge"]:
        topics = v2_state["knowledge"]["topics"]
        for name, topic_data in topics.items():
            # 确保有所有 v2 字段
            topic_data.setdefault("parents", [])
            topic_data.setdefault("relations", [])
            topic_data.setdefault("explored_by", [])
            topic_data.setdefault("fully_explored", False)
            topic_data.setdefault("schema_version", "2.0")
            
            # 迁移：如果有 parent 字段（字符串），转为 parents[]
            if "parent" in topic_data and topic_data["parent"]:
                parent = topic_data.pop("parent")
                if parent not in topic_data["parents"]:
                    topic_data["parents"].append(parent)
    
    return v2_state

def detect_schema_version(state: dict) -> str:
    """检测 state 的 schema 版本"""
    return state.get("schema_version", "1.0")
```

- [ ] 单 parent 正确转为 parents[]
- [ ] 不丢失现有数据
- [ ] 迁移可重复执行（幂等）

```python
def add_relation(from_topic: str, to_topic: str, 
                 relation_type: str = "associated") -> None:
    """
    添加关系（多父节点支持）
    
    Args:
        from_topic: 源 topic
        to_topic: 目标 topic
        relation_type: 关系类型 (associated | parent_child | decomposed_from | related_to)
    """
    state = _load_state()
    topics = state["knowledge"]["topics"]
    now = datetime.now(timezone.utc).isoformat()
    
    # 确保源 topic 存在
    if from_topic not in topics:
        topics[from_topic] = _create_default_topic(from_topic)
    
    # 确保目标 topic 存在
    if to_topic not in topics:
        topics[to_topic] = _create_default_topic(to_topic)
    
    # 更新 children / parents（双向）
    if to_topic not in topics[from_topic].get("children", []):
        topics[from_topic].setdefault("children", []).append(to_topic)
    
    if from_topic not in topics[to_topic].get("parents", []):
        topics[to_topic].setdefault("parents", []).append(from_topic)
    
    # 添加详细关系记录
    relation = {
        "from": from_topic,
        "to": to_topic,
        "type": relation_type,
        "created_at": now,
    }
    topics[from_topic].setdefault("relations", []).append(relation)
    
    _save_state(state)

def should_explore(topic: str, from_parent: str | None = None) -> tuple[bool, str]:
    """
    判断是否需要探索该节点
    
    Returns:
        (should_explore, reason)
        - (True, "first_visit"): 新节点
        - (True, "not_yet_explored"): 已存在但未探索
        - (False, "linked_only"): 已探索，只更新关联
        - (False, "already_explored"): 已探索且已知此父节点
    """
    state = _load_state()
    topics = state["knowledge"]["topics"]
    
    if topic not in topics:
        return True, "first_visit"
    
    topic_data = topics[topic]
    
    if not topic_data.get("known") and not topic_data.get("explored"):
        return True, "not_yet_explored"
    
    # 已探索，检查是否需要更新关联
    if from_parent:
        explored_by = topic_data.get("explored_by", [])
        parents = topic_data.get("parents", [])
        
        if from_parent not in explored_by and from_parent not in parents:
            # 新父节点，更新关联
            topic_data.setdefault("parents", []).append(from_parent)
            topic_data.setdefault("explored_by", []).append(from_parent)
            _save_state(state)
            return False, "linked_only"
    
    return False, "already_explored"

def mark_fully_explored(topic: str) -> None:
    """标记节点为充分探索"""
    state = _load_state()
    topics = state["knowledge"]["topics"]
    
    if topic in topics:
        topics[topic]["fully_explored"] = True
        topics[topic]["status"] = "complete"
        topics[topic]["fully_explored_at"] = datetime.now(timezone.utc).isoformat()
        _save_state(state)

def get_high_degree_unexplored() -> str | None:
    """
    获取度（入度 + 出度）最高且未充分探索的节点
    
    Returns:
        节点名称，如果没有则返回 None
    """
    state = _load_state()
    topics = state["knowledge"]["topics"]
    
    candidates = []
    for name, data in topics.items():
        if data.get("fully_explored"):
            continue
        
        degree = len(data.get("parents", [])) + len(data.get("children", []))
        candidates.append((degree, name))
    
    if not candidates:
        return None
    
    # 按度降序，度相同按名称排序（确定性选择）
    candidates.sort(key=lambda x: (-x[0], x[1]))
    return candidates[0][1]

def get_topic_graph(topic: str) -> dict | None:
    """
    获取节点的完整图结构信息
    
    Returns:
        {
            "parents": [...],
            "children": [...],
            "relations": [...],
            "explored_by": [...],
            "degree": int,
        }
    """
    state = _load_state()
    topics = state["knowledge"]["topics"]
    
    if topic not in topics:
        return None
    
    data = topics[topic]
    return {
        "parents": data.get("parents", []),
        "children": data.get("children", []),
        "relations": data.get("relations", []),
        "explored_by": data.get("explored_by", []),
        "degree": len(data.get("parents", [])) + len(data.get("children", [])),
    }

# ========== 向后兼容适配（修改现有方法）==========

def add_child(parent: str, child: str) -> None:
    """
    保留向后兼容，内部调用 add_relation
    """
    add_relation(parent, child, "parent_child")
    
    # 保持原有 children 字段更新逻辑（冗余但兼容）
    state = _load_state()
    topics = state["knowledge"]["topics"]
    
    if parent in topics:
        if child not in topics[parent].get("children", []):
            topics[parent].setdefault("children", []).append(child)
        _save_state(state)
```

```python
import json
import os
from datetime import datetime, timezone
from core.models.topic import Topic
from core.models.migration import migrate_state_v1_to_v2, detect_schema_version
from core.repository.base import KnowledgeRepository

class JSONKnowledgeRepository(KnowledgeRepository):
    """JSON 文件存储实现"""
    
    def __init__(self, path: str = "knowledge/state.json"):
        self.path = path
        self._cache: dict[str, Topic] = {}
        self._load()
    
    def _load(self) -> None:
        """加载数据"""
        if not os.path.exists(self.path):
            self._state = self._create_default_state()
            return
        
        with open(self.path, "r", encoding="utf-8") as f:
            raw_state = json.load(f)
        
        # Schema 迁移
        version = detect_schema_version(raw_state)
        if version == "1.0":
            raw_state = migrate_state_v1_to_v2(raw_state)
        
        self._state = raw_state
        
        # 加载到缓存
        self._cache = {}
        for name, data in self._state.get("knowledge", {}).get("topics", {}).items():
            self._cache[name] = Topic.from_dict(data)
    
    def _save(self) -> None:
        """保存数据"""
        # 从缓存更新 state
        for name, topic in self._cache.items():
            self._state["knowledge"]["topics"][name] = topic.to_dict()
        
        self._state["last_update"] = datetime.now(timezone.utc).isoformat()
        
        os.makedirs(os.path.dirname(self.path), exist_ok=True)
        with open(self.path, "w", encoding="utf-8") as f:
            json.dump(self._state, f, ensure_ascii=False, indent=2)
    
    def get_topic(self, name: str) -> Topic | None:
        return self._cache.get(name)
    
    def save_topic(self, topic: Topic) -> None:
        self._cache[topic.name] = topic
        self._save()
    
    def get_all_topics(self) -> list[Topic]:
        return list(self._cache.values())
    
    def add_relation(self, from_topic: str, to_topic: str,
                     relation_type: str = "associated") -> None:
        # 确保两个 topic 都存在
        if from_topic not in self._cache:
            self._cache[from_topic] = Topic(name=from_topic)
        if to_topic not in self._cache:
            self._cache[to_topic] = Topic(name=to_topic)
        
        # 更新关系
        from_t = self._cache[from_topic]
        to_t = self._cache[to_topic]
        
        from_t.add_child(to_topic)
        to_t.add_parent(from_topic)
        
        from_t.relations.append(Relation(
            from_topic=from_topic,
            to_topic=to_topic,
            relation_type=relation_type,
        ))
        
        self._save()
    
    def get_neighbors(self, topic: str,
                      relation_type: str | None = None) -> list[str]:
        t = self._cache.get(topic)
        if not t:
            return []
        
        if relation_type:
            # 按关系类型过滤
            return [
                r.to_topic for r in t.relations
                if r.relation_type == relation_type
            ]
        
        return t.children + t.parents
    
    def get_high_degree_unexplored(self) -> str | None:
        candidates = [
            (t.degree, name)
            for name, t in self._cache.items()
            if not t.fully_explored
        ]
        
        if not candidates:
            return None
        
        candidates.sort(key=lambda x: (-x[0], x[1]))
        return candidates[0][1]
    
    def get_storage_path(self) -> str:
        return self.path
    
    def _create_default_state(self) -> dict:
        return {
            "version": "1.0",
            "schema_version": "2.0",
            "last_update": None,
            "knowledge": {"topics": {}},
            "curiosity_queue": [],
            "exploration_log": [],
            "config": {
                "curiosity_top_k": 3,
                "max_knowledge_nodes": 100,
                "notification_threshold": 7.0,
            },
        }
```

```python
from dataclasses import dataclass, field
from typing import Literal
import yaml
import os

@dataclass
class SpiderConfig:
    """Spider 配置"""
    
    # 运行参数
    loop_interval: int = 30
    max_consecutive_low_gain: int = 3
    min_marginal_return: float = 0.3
    high_quality_threshold: float = 7.0
    default_exploration_depth: Literal["shallow", "medium", "deep"] = "medium"
    
    # 策略选择
    frontier_strategy: Literal["high_degree", "breadth_first", "depth_first"] = "high_degree"
    quality_evaluator: Literal["v2", "v1"] = "v2"
    
    # 路径配置
    kg_storage_path: str = "knowledge/state.json"
    checkpoint_path: str = "state/spider_state.json"
    discoveries_dir: str = "shared_knowledge/ca/discoveries"
    propositions_dir: str = "shared_knowledge/r1d3/propositions"
    
    # EventBus 配置
    event_bus_type: Literal["memory", "file", "redis"] = "memory"
    event_bus_path: str | None = None  # 用于 file 类型
    
    @classmethod
    def from_file(cls, path: str = "config/spider.yaml") -> "SpiderConfig":
        """从 YAML 文件加载配置"""
        if not os.path.exists(path):
            return cls()  # 返回默认配置
        
        with open(path, "r", encoding="utf-8") as f:
            data = yaml.safe_load(f)
        
        return cls(**data.get("spider", {}))
    
    @classmethod
    def from_env(cls) -> "SpiderConfig":
        """从环境变量加载配置（覆盖文件配置）"""
        config = cls.from_file()
        
        # 环境变量覆盖
        if "SPIDER_LOOP_INTERVAL" in os.environ:
            config.loop_interval = int(os.environ["SPIDER_LOOP_INTERVAL"])
        if "SPIDER_MIN_MARGINAL_RETURN" in os.environ:
            config.min_marginal_return = float(os.environ["SPIDER_MIN_MARGINAL_RETURN"])
        
        return config
```
