# Curious Agent v0.2.4 TDD 实施方案

> **方法论**: Test-Driven Development (TDD)  
> **循环**: Red → Green → Refactor  
> **铁律**: 没有失败的测试，不写生产代码

---

## 一、TDD 工作流

### 1.1 开发顺序

每个组件按以下顺序开发：

```
Test 1 (RED) → Implementation 1 (GREEN) → Refactor
      ↓
Test 2 (RED) → Implementation 2 (GREEN) → Refactor
      ↓
Test 3 (RED) → Implementation 3 (GREEN) → Refactor
      ↓
Integration Tests → Refactor Architecture
```

### 1.2 测试金字塔

```
         /\
        /  \     E2E Tests (少量)
       /____\      - 完整探索流程
      /      \     - Kill/恢复测试
     /________\    
    /          \   Integration Tests (中等)
   /____________\    - Repository + KG
  /              \   - Spider + Components
 /________________\  
/                  \ Unit Tests (大量)
/____________________\ - Topic Model
                       - Schema Migration
                       - Strategies
```

```python
# core/models/topic.py
from dataclasses import dataclass, field
from datetime import datetime

@dataclass
class Topic:
    name: str
    known: bool = False
    depth: float = 0.0
    summary: str = ""
    sources: list = field(default_factory=list)
    status: str = "unexplored"
    explore_count: int = 0
    marginal_returns: list = field(default_factory=list)
    last_quality: float = 0.0
    parents: list = field(default_factory=list)
    children: list = field(default_factory=list)
    relations: list = field(default_factory=list)
    explored_by: list = field(default_factory=list)
    fully_explored: bool = False
    created_at: datetime = field(default_factory=datetime.now)
    explored_at: datetime = None
    fully_explored_at: datetime = None
    schema_version: str = "2.0"
    
    def to_dict(self) -> dict:
        return {
            "name": self.name,
            "known": self.known,
            "depth": self.depth,
            "summary": self.summary,
            "sources": self.sources,
            "status": self.status,
            "explore_count": self.explore_count,
            "marginal_returns": self.marginal_returns,
            "last_quality": self.last_quality,
            "parents": self.parents,
            "children": self.children,
            "relations": self.relations,
            "explored_by": self.explored_by,
            "fully_explored": self.fully_explored,
            "created_at": self.created_at.isoformat() if self.created_at else None,
            "explored_at": self.explored_at.isoformat() if self.explored_at else None,
            "fully_explored_at": self.fully_explored_at.isoformat() if self.fully_explored_at else None,
            "schema_version": self.schema_version,
        }
    
    @classmethod
    def from_dict(cls, data: dict):
        return cls(
            name=data["name"],
            known=data.get("known", False),
            depth=data.get("depth", 0.0),
            summary=data.get("summary", ""),
            sources=data.get("sources", []),
            status=data.get("status", "unexplored"),
            explore_count=data.get("explore_count", 0),
            marginal_returns=data.get("marginal_returns", []),
            last_quality=data.get("last_quality", 0.0),
            parents=data.get("parents", []),
            children=data.get("children", []),
            relations=data.get("relations", []),
            explored_by=data.get("explored_by", []),
            fully_explored=data.get("fully_explored", False),
            created_at=datetime.fromisoformat(data["created_at"]) if data.get("created_at") else datetime.now(),
            explored_at=datetime.fromisoformat(data["explored_at"]) if data.get("explored_at") else None,
            fully_explored_at=datetime.fromisoformat(data["fully_explored_at"]) if data.get("fully_explored_at") else None,
            schema_version=data.get("schema_version", "2.0"),
        )
```

```python
# 在 Topic 类中添加方法

@dataclass
class Topic:
    # ... 原有字段 ...
    explored: bool = False  # 新增字段
    
    def add_parent(self, parent: str) -> None:
        """添加父节点（去重）"""
        if parent not in self.parents:
            self.parents.append(parent)
    
    def add_child(self, child: str) -> None:
        """添加子节点（去重）"""
        if child not in self.children:
            self.children.append(child)
    
    def mark_explored(self, by: str | None = None) -> None:
        """标记为已探索"""
        self.known = True
        self.explored = True
        self.status = "partial"
        self.explored_at = datetime.now()
        if by and by not in self.explored_by:
            self.explored_by.append(by)
    
    def mark_fully_explored(self) -> None:
        """标记为充分探索"""
        self.fully_explored = True
        self.status = "complete"
        self.fully_explored_at = datetime.now()
    
    @property
    def degree(self) -> int:
        """节点的度（入度 + 出度）"""
        return len(self.parents) + len(self.children)
```

**实现（GREEN）**:
```python
def migrate_state_v1_to_v2(v1_state: dict) -> dict:
    """
    迁移 v1 state.json 到 v2 格式
    """
    v2_state = v1_state.copy()
    v2_state["schema_version"] = "2.0"
    
    # 迁移 topics
    if "knowledge" in v2_state and "topics" in v2_state["knowledge"]:
        topics = v2_state["knowledge"]["topics"]
        for name, topic_data in topics.items():
            # 确保有所有 v2 字段
            topic_data.setdefault("parents", [])
            topic_data.setdefault("relations", [])
            topic_data.setdefault("explored_by", [])
            topic_data.setdefault("fully_explored", False)
            topic_data.setdefault("schema_version", "2.0")
            
            # 迁移：如果有 parent 字段（字符串），转为 parents[]
            if "parent" in topic_data and topic_data["parent"]:
                parent = topic_data.pop("parent")
                if parent not in topic_data["parents"]:
                    topic_data["parents"].append(parent)
    
    return v2_state
```

```python
        pass
    
    @abstractmethod
    def save_topic(self, topic: Topic) -> None:
        pass
    
    @abstractmethod
    def get_all_topics(self) -> list[Topic]:
        pass
    
    @abstractmethod
    def add_relation(self, from_topic: str, to_topic: str,
                     relation_type: str = "associated") -> None:
        pass
    
    @abstractmethod
    def get_neighbors(self, topic: str,
                      relation_type: str | None = None) -> list[str]:
        pass
    
    @abstractmethod
```

```python
# core/repository/json_repository.py

import json
import os
from datetime import datetime, timezone
from core.models.topic import Topic
from core.models.migration import migrate_state_v1_to_v2, detect_schema_version
from core.repository.base import KnowledgeRepository

class JSONKnowledgeRepository(KnowledgeRepository):
    def __init__(self, path: str = "knowledge/state.json"):
        self.path = path
        self._cache: dict[str, Topic] = {}
        self._load()
    
    def _load(self) -> None:
        if not os.path.exists(self.path):
            self._state = self._create_default_state()
            return
        
        with open(self.path, "r", encoding="utf-8") as f:
            raw_state = json.load(f)
        
        version = detect_schema_version(raw_state)
        if version == "1.0":
            raw_state = migrate_state_v1_to_v2(raw_state)
        
        self._state = raw_state
        
        # 加载到缓存
        self._cache = {}
        for name, data in self._state.get("knowledge", {}).get("topics", {}).items():
            self._cache[name] = Topic.from_dict(data)
    
    def _save(self) -> None:
        for name, topic in self._cache.items():
            self._state["knowledge"]["topics"][name] = topic.to_dict()
        
        self._state["last_update"] = datetime.now(timezone.utc).isoformat()
        
        os.makedirs(os.path.dirname(self.path), exist_ok=True)
        with open(self.path, "w", encoding="utf-8") as f:
            json.dump(self._state, f, ensure_ascii=False, indent=2)
    
    def get_topic(self, name: str) -> Topic | None:
        return self._cache.get(name)
    
    def save_topic(self, topic: Topic) -> None:
        self._cache[topic.name] = topic
        self._save()
    
    def get_all_topics(self) -> list[Topic]:
        return list(self._cache.values())
    
    def add_relation(self, from_topic: str, to_topic: str,
                     relation_type: str = "associated") -> None:
        if from_topic not in self._cache:
            self._cache[from_topic] = Topic(name=from_topic)
        if to_topic not in self._cache:
            self._cache[to_topic] = Topic(name=to_topic)
        
        from_t = self._cache[from_topic]
        to_t = self._cache[to_topic]
        
        from_t.add_child(to_topic)
        to_t.add_parent(from_topic)
        
        self._save()
    
    def get_neighbors(self, topic: str, relation_type: str | None = None) -> list[str]:
        t = self._cache.get(topic)
        if not t:
            return []
        return t.children + t.parents
    
    def get_high_degree_unexplored(self) -> str | None:
        candidates = [
            (t.degree, name)
            for name, t in self._cache.items()
            if not t.fully_explored
        ]
        
        if not candidates:
            return None
        
        candidates.sort(key=lambda x: (-x[0], x[1]))
        return candidates[0][1]
    
    def _create_default_state(self) -> dict:
        return {
            "version": "1.0",
            "schema_version": "2.0",
            "last_update": None,
            "knowledge": {"topics": {}},
            "curiosity_queue": [],
            "exploration_log": [],
            "config": {
                "curiosity_top_k": 3,
                "max_knowledge_nodes": 100,
                "notification_threshold": 7.0,
            },
        }
```
