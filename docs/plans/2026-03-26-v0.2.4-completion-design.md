## 阶段 A: InsightSynthesizer Layer 3

### 目标
实现跨子话题的原创分析合成能力

### 组件设计

#### A1. InsightSynthesizer 核心类

**文件**: `core/insight_synthesizer.py` (新建)

```python
from dataclasses import dataclass
from typing import Optional

@dataclass
class Insight:
    id: str
    topic: str
    hypothesis: str
    type: str           # causal|correlation|contrast|deduction
    reasoning: str
    confidence: float   # 0-1
    supporting_snippets: list[str]
    generated_by: str   # "InsightSynthesizer"
    timestamp: str

@dataclass  
class Pattern:
    pattern: str
    supporting_snippets: list[str]
    related_sub_topics: list[str]

@dataclass
class Hypothesis:
    hypothesis: str
    type: str
    reasoning: str
    supporting_snippets: list[str]


class InsightSynthesizer:
    """
    第三层：跨子话题原始分析合成
    
    输入: exploration_results[sub_topic] = [SearchResult, ...]
    输出: insights = [Insight, ...]
    """
    
    def __init__(self, llm_client=None):
        self.llm = llm_client or LLMClient()
    
    def synthesize(self, topic: str, sub_topic_results: dict) -> list[Insight]:
        """
        合成入口
        
        Args:
            topic: 主话题
            sub_topic_results: {sub_topic: [search_results]}
            
        Returns:
            list[Insight]: 原创洞察列表
        """
        # 1. 聚合所有子话题的搜索片段
        all_snippets = []
        for sub_topic, results in sub_topic_results.items():
            all_snippets.extend(self._extract_snippets(results))
        
        # 2. 跨维度模式识别
        patterns = self.cross_topic_patterns(topic, all_snippets)
        
        # 3. 基于模式生成原创假设
        hypotheses = self.generate_hypotheses(patterns)
        
        # 4. 置信度评分 + 过滤
        scored_hypotheses = []
        for h in hypotheses:
            confidence = self.compute_confidence(h, all_snippets)
            if confidence >= 0.5:  # 只保留置信度 >= 0.5
                scored_hypotheses.append({**h, "confidence": confidence})
        
        # 5. 格式化为 Insight
        return [self._format_insight(h, topic) for h in scored_hypotheses]
    
    def cross_topic_patterns(self, topic: str, snippets: list[str]) -> list[Pattern]:
        """
        跨子话题模式识别
        
        Example:
            agent 的 memory + planning + tool use 三个子话题
            探索结果中共同出现的模式：
            - "context window" 同时出现在 memory 和 planning
            - "tool calling" 与 planning 的 "reasoning" 经常共同出现
            → 生成假设：context window 是连接 memory 和 planning 的关键桥梁
        """
        prompt = f"""
给定主题：{topic}
收集到的信息片段：
{self._format_snippets(snippets)}

任务：从这些片段中识别跨维度的模式。
输出 JSON 格式：
{{
  "patterns": [
    {{
      "pattern": "模式描述",
      "supporting_snippets": ["支撑片段1", "支撑片段2"],
      "related_sub_topics": ["相关的子话题列表"]
    }}
  ]
}}
"""
        response = self.llm.chat(prompt)
        return self._parse_patterns(response)
    
    def generate_hypotheses(self, patterns: list[Pattern]) -> list[Hypothesis]:
        """
        基于识别的模式生成原创假设
        
        假设类型：
        - causal: A → B（因为A，所以B）
        - correlation: A ⟷ B（A和B相关，但因果不明）
        - contrast: A vs B（A和B有本质差异）
        - deduction: A → B → C（链式推演）
        """
        prompt = f"""
基于以下跨维度模式，生成原创假设：

{self._format_patterns(patterns)}

要求：
1. 每个假设必须有推理过程（reasoning）
2. 假设应该超出原始信息的简单总结
3. 允许有一定推测性，但必须有片段支撑
4. 生成 3-5 个最具洞察力的假设

输出 JSON 格式：
{{
  "hypotheses": [
    {{
      "hypothesis": "假设内容",
      "type": "causal|correlation|contrast|deduction",
      "reasoning": "推理过程"
    }}
  ]
}}
"""
        response = self.llm.chat(prompt)
        return self._parse_hypotheses(response)
    
    def compute_confidence(self, hypothesis: Hypothesis, snippets: list[str]) -> float:
        """
        计算假设置信度
        
        评分维度：
        - 支持片段数量（越多越可信）
        - 片段来源多样性（来源越多越可信）
        - 假设逻辑一致性（推理链是否自洽）
        """
        support_count = len(hypothesis.supporting_snippets)
        diversity = self._compute_source_diversity(hypothesis.supporting_snippets)
        consistency = self._check_consistency(hypothesis)
        
        confidence = (
            min(support_count / 5, 1.0) * 0.4 +   # 片段数量
            diversity * 0.3 +                       # 来源多样性
            consistency * 0.3                       # 一致性
        )
        return round(confidence, 2)
    
    def _format_insight(self, hypothesis: dict, topic: str) -> Insight:
        """格式化为标准 Insight Object"""
        import uuid
        from datetime import datetime
        
        return Insight(
            id=f"ins_{datetime.now().strftime('%Y%m%d')}_{uuid.uuid4().hex[:6]}",
            topic=topic,
            hypothesis=hypothesis["hypothesis"],
            type=hypothesis["type"],
            reasoning=hypothesis["reasoning"],
            confidence=hypothesis["confidence"],
            supporting_snippets=hypothesis.get("supporting_snippets", []),
            generated_by="InsightSynthesizer",
            timestamp=datetime.now().isoformat()
        )
```

#### A2. 集成到探索流程

**修改**: `spider_engine.py` 或新建 `core/exploration_pipeline.py`

```python
class ExplorationPipeline:
    """三层探索管道"""
    
    def __init__(self):
        self.decomposer = CuriosityDecomposer(...)
        self.explorer = Explorer(...)
        self.synthesizer = InsightSynthesizer(...)
    
    async def explore_topic(self, topic: str) -> dict:
        """完整的三层探索流程"""
        # Layer 1: 分解
        sub_topics = await self.decomposer.decompose(topic)
        
        # Layer 2: 探索
        sub_topic_results = {}
        for sub in sub_topics:
            results = await self.explorer.explore(sub)
            sub_topic_results[sub] = results
        
        # Layer 3: 合成（核心新增）
        insights = self.synthesizer.synthesize(topic, sub_topic_results)
        
        return {
            "topic": topic,
            "sub_topic_results": sub_topic_results,
            "insights": insights  # 新增
        }
```

```python
from typing import Optional
from core.models.topic import Topic


class KGGraph:
    """
    知识图谱图结构管理器
    
    支持多父节点，避免重复探索
    """
    
    def __init__(self, repo: KnowledgeRepository):
        self.repo = repo
    
    def should_explore(self, node: str, from_parent: Optional[str] = None) -> tuple[bool, str]:
        """
        判断是否需要探索节点
        
        Returns:
            (True, "first_visit")          → 新节点，需要探索
            (True, "not_yet_explored")     → 已存在但未探索，需要探索
            (False, "linked_only")         → 已探索，只更新关联
            (False, "already_explored")    → 已探索且已知此父节点，跳过
        """
        topic = self.repo.get_topic(node)
        
        if not topic:
            return True, "first_visit"
        
        if not topic.explored:
            return True, "not_yet_explored"
        
        # 已探索过
        if from_parent and from_parent in topic.parents:
            return False, "already_explored"
        
        return False, "linked_only"
    
    def add_relation(self, from_topic: str, to_topic: str, relation_type: str = "associated"):
        """添加关系，双向更新 parents/children"""
        self.repo.add_relation(from_topic, to_topic, relation_type)
    
    def get_high_degree_unexplored(self) -> Optional[str]:
        """返回度最高的未充分探索节点"""
        return self.repo.get_high_degree_unexplored()
```
