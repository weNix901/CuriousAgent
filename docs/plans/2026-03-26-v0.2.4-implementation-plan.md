# core/api/r1d3_tools.py
"""R1D3 Tool Handler - API for R1D3 to interact with Curious Agent"""
from typing import Optional
from core.repository.base import KnowledgeRepository
from core.models.topic import Topic


class R1D3ToolHandler:
    """
    Handles R1D3 Tool calls for CA integration
    
    Tools:
    - curious_check_confidence: Query topic confidence level
    - curious_agent_inject: Trigger directed exploration
    """
    
    def __init__(self, repo: Optional[KnowledgeRepository] = None):
        from core.repository.json_repository import JSONKnowledgeRepository
        self.repo = repo or JSONKnowledgeRepository()
    
    def curious_check_confidence(self, topic: str) -> dict:
        """
        R1D3 queries confidence level for a topic
        
        Returns:
            {
                "topic": str,
                "confidence": float,      # 0-1
                "level": str,             # novice|competent|proficient|expert
                "explore_count": int,
                "gaps": list
            }
        """
        t = self.repo.get_topic(topic)
        
        if not t or not t.explored:
            return {
                "topic": topic,
                "confidence": 0.0,
                "level": "novice",
                "explore_count": 0,
                "gaps": ["No exploration data available"]
            }
        
        # Calculate confidence based on explore_count and quality
        confidence = self._calculate_confidence(t)
        level = self._confidence_to_level(confidence)
        
        return {
            "topic": topic,
            "confidence": confidence,
            "level": level,
            "explore_count": t.explore_count,
            "gaps": self._identify_gaps(t)
        }
    
    def _calculate_confidence(self, topic: Topic) -> float:
        """Calculate confidence score 0-1 based on exploration metrics"""
        base_confidence = min(topic.explore_count / 5, 1.0) * 0.6
        quality_boost = (topic.last_quality or 0) / 10 * 0.4
        return min(base_confidence + quality_boost, 1.0)
    
    def _confidence_to_level(self, confidence: float) -> str:
        """Convert confidence to level"""
        if confidence < 0.3:
            return "novice"
        elif confidence < 0.6:
            return "competent"
        elif confidence < 0.85:
            return "proficient"
        else:
            return "expert"
    
    def _identify_gaps(self, topic: Topic) -> list:
        """Identify knowledge gaps"""
        gaps = []
        if topic.explore_count < 3:
            gaps.append("Limited exploration depth")
        if not topic.fully_explored:
            gaps.append("Topic not fully explored")
        return gaps
```

# Add to core/api/r1d3_tools.py

    def curious_agent_inject(self, topic: str, context: str = "", 
                            depth: str = "medium", source: str = "r1d3") -> dict:
        """
        R1D3 triggers directed exploration
        
        Args:
            topic: Topic to explore
            context: Original user question (for scoring)
            depth: Exploration depth (shallow|medium|deep)
            source: Injection source (default: r1d3)
            
        Returns:
            {
                "status": "success"|"error",
                "topic_id": str,
                "queue_position": int,
                "priority": bool
            }
        """
        config = self._get_config()
        priority_config = config.get("injection_priority", {})
        
        # Check if priority processing enabled
        if priority_config.get("enabled", False):
            if source in priority_config.get("priority_sources", []):
                return self._inject_with_priority(topic, context, depth, priority_config)
        
        # Normal injection
        return self._inject_to_queue(topic, context, depth)
    
    def _inject_with_priority(self, topic: str, context: str, depth: str, 
                             priority_config: dict) -> dict:
        """Inject with priority boost"""
        boost_score = priority_config.get("boost_score", 2.0)
        
        # Add to queue with boosted score
        # This is a simplified version - in production would use proper queue
        return {
            "status": "success",
            "topic_id": f"topic_{hash(topic) % 10000}",
            "queue_position": 1,  # Priority = front of queue
            "priority": True,
            "boosted_score": 5.0 + boost_score
        }
    
    def _inject_to_queue(self, topic: str, context: str, depth: str) -> dict:
        """Normal queue injection"""
        # Simplified - would integrate with actual queue system
        return {
            "status": "success",
            "topic_id": f"topic_{hash(topic) % 10000}",
            "queue_position": -1,  # Unknown without real queue
            "priority": False
        }
    
    def _get_config(self) -> dict:
        """Get CA configuration"""
        try:
            from core.config.spider_config import SpiderConfig
            config = SpiderConfig()
            return config.__dict__ if hasattr(config, '__dict__') else {}
        except:
            return {}

# core/sync/r1d3_sync.py
"""R1D3 Sync - Manage discovery sharing between CA and R1D3"""
import json
import os
from pathlib import Path
from typing import list, dict


class R1D3Sync:
    """
    Manages synchronization of discoveries to R1D3
    
    Features:
    - Mark discoveries as shared
    - Get unshared discoveries for R1D3 to consume
    - Track shared status with 'shared' field
    """
    
    DEFAULT_DISCOVERIES_DIR = "shared_knowledge/ca/discoveries"
    
    def __init__(self, discoveries_dir: str = None):
        self.discoveries_dir = Path(discoveries_dir or self.DEFAULT_DISCOVERIES_DIR)
        self.discoveries_dir.mkdir(parents=True, exist_ok=True)
    
    def mark_discovery_shared(self, discovery_filename: str) -> bool:
        """
        Mark a discovery as shared with R1D3
        
        Args:
            discovery_filename: Name of the discovery file
            
        Returns:
            True if successfully marked, False otherwise
        """
        file_path = self.discoveries_dir / discovery_filename
        
        if not file_path.exists():
            return False
        
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                data = json.load(f)
            
            # Add shared field
            data['shared'] = True
            data['shared_at'] = self._get_timestamp()
            
            with open(file_path, 'w', encoding='utf-8') as f:
                json.dump(data, f, indent=2, ensure_ascii=False)
            
            return True
        except Exception as e:
            print(f"[R1D3Sync] Error marking discovery: {e}")
            return False
    
    def get_unshared_discoveries(self) -> list[dict]:
        """
        Get all unshared discoveries
        
        Returns:
            List of discovery dictionaries with shared=False or no shared field
        """
        unshared = []
        
        try:
            for filename in os.listdir(self.discoveries_dir):
                if not filename.endswith('.json'):
                    continue
                
                file_path = self.discoveries_dir / filename
                
                try:
                    with open(file_path, 'r', encoding='utf-8') as f:
                        data = json.load(f)
                    
                    # Check if not shared
                    if not data.get('shared', False):
                        data['_filename'] = filename  # Add filename for reference
                        unshared.append(data)
                
                except json.JSONDecodeError:
                    continue
                except Exception as e:
                    print(f"[R1D3Sync] Error reading {filename}: {e}")
                    continue
        
        except FileNotFoundError:
            pass
        
        return unshared
    
    def get_shared_discoveries(self, since: str = None) -> list[dict]:
        """
        Get discoveries shared since a specific time
        
        Args:
            since: ISO timestamp to filter discoveries shared after this time
            
        Returns:
            List of shared discovery dictionaries
        """
        shared = []
        
        try:
            for filename in os.listdir(self.discoveries_dir):
                if not filename.endswith('.json'):
                    continue
                
                file_path = self.discoveries_dir / filename
                
                try:
                    with open(file_path, 'r', encoding='utf-8') as f:
                        data = json.load(f)
                    
                    if data.get('shared', False):
                        if since is None or data.get('shared_at', '') > since:
                            shared.append(data)
                
                except Exception:
                    continue
        
        except FileNotFoundError:
            pass
        
        return shared
    
    def _get_timestamp(self) -> str:
        """Get current ISO timestamp"""
        from datetime import datetime, timezone
        return datetime.now(timezone.utc).isoformat()
```
