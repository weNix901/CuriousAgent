## 一、问题总览

### 1.1 原规划问题整合

原 `next_move_v0.2.7.md` 识别的 5 个根因 (G1-G5) 及其在新架构中的解决方案：

| 根因 | 原问题 | 症状 | 新架构解决方案 |
|------|--------|------|---------------|
| G1 | `claim_pending_item()` 缺失 | 队列无人消费 | `QueueService.claim_next()` 原子操作 |
| G2 | `_explore_in_thread` 缺少 `mark_topic_done` | 状态不更新 | `StateMachine` 统一管理状态转换 |
| G3 | `add_exploration_result` typo + 逻辑颠倒 | parent 关系错误 | `LineageInfo` 携带明确 parent 信息 |
| G4 | `knowledge_graph=None` 传入 QualityV2 | quality=0 | 懒加载模式，自动获取 KG 实例 |
| G5 | `api_inject` 无 parent link 写入 | 父子关系丢失 | `LineageInfo` 在入队时即携带完整血缘 |

### 1.2 系统性问题解决方案

除原 5 个根因外，新架构还解决以下系统性问题：

1. **状态管理碎片化** → 统一 `ExplorationStateMachine`
2. **数据不一致** → `ConsistencyMonitor` 实时监控
3. **状态转换不可追溯** → 完整的状态历史记录
4. **并发安全隐患** → 原子 claim 机制 + 状态超时
5. **调试困难** → 完整血缘信息便于追踪

### 2.2 核心设计决策

#### 决策 1: 统一状态机

**决策**: 所有状态转换必须通过 `ExplorationStateMachine`

**理由**:
- 消除分散的状态更新逻辑
- 确保状态转换合法性
- 提供完整的状态历史
- 便于调试和监控

#### 决策 2: 数据血缘追踪

**决策**: 每个队列项必须携带 `LineageInfo`

**理由**:
- 解决 parent link 传递问题
- 支持完整的探索路径追溯
- 便于分析话题来源和演化

#### 决策 3: Service Layer 抽象

**决策**: 新增 Service Layer 作为 Agent 和 Data 的中间层

**理由**:
- 解耦业务逻辑和数据存储
- 便于单元测试
- 支持未来更换存储实现

```python
from datetime import datetime
from typing import List, Optional, Dict, Callable
from threading import Lock

class State(Enum):
    """探索状态枚举"""
    PENDING = "pending"         # 等待处理
    CLAIMED = "claimed"         # 已被claim，准备探索
    EXPLORING = "exploring"     # 正在探索

    PAUSED = "paused"           # 暂停（可恢复）
    FAILED = "failed"           # 探索失败
    CANCELLED = "cancelled"     # 已取消

class Actor(Enum):
    """状态转换执行者"""
    SYSTEM = "system"
    SPIDER_AGENT = "spider_agent"
    DREAM_AGENT = "dream_agent"
    ASYNC_TRIGGER = "async_trigger"
    USER = "user"

@dataclass
class StateTransition:
    """状态转换记录"""
    from_state: Optional[State]
    to_state: State
    actor: Actor
    timestamp: datetime = field(default_factory=datetime.now)
    reason: str = ""
    context: Dict = field(default_factory=dict)

@dataclass
class StateConfig:
    """状态配置"""

    valid_actors: List[Actor]
    timeout_seconds: Optional[int] = None
    max_retries: int = 0
    auto_retry: bool = False
```

#### 3.1.2 状态机实现

```python
class ExplorationStateMachine:
    """
    统一的状态机管理所有探索状态转换
    
    线程安全，支持并发访问
    """
    
    # 状态配置表
    STATE_CONFIG = {
        State.PENDING: StateConfig(
            allowed_transitions=[State.CLAIMED, State.CANCELLED],
            valid_actors=[Actor.SYSTEM, Actor.USER],
            timeout_seconds=None,
        ),
        State.CLAIMED: StateConfig(
            allowed_transitions=[State.EXPLORING, State.PENDING],
            valid_actors=[Actor.SPIDER_AGENT, Actor.ASYNC_TRIGGER],
            timeout_seconds=300,  # 5分钟超时自动释放
        ),
        State.EXPLORING: StateConfig(
            allowed_transitions=[State.DONE, State.PAUSED, State.FAILED],
            valid_actors=[Actor.SPIDER_AGENT, Actor.ASYNC_TRIGGER],
            timeout_seconds=1800,  # 30分钟超时
        ),
        State.DONE: StateConfig(
            allowed_transitions=[],
            valid_actors=[Actor.SYSTEM],
        ),
        State.PAUSED: StateConfig(
            allowed_transitions=[State.PENDING],
            valid_actors=[Actor.SYSTEM],
            auto_retry=True,
        ),
        State.FAILED: StateConfig(
            allowed_transitions=[State.PENDING],
            valid_actors=[Actor.SYSTEM],
            max_retries=3,
        ),
        State.CANCELLED: StateConfig(
            allowed_transitions=[],
            valid_actors=[Actor.SYSTEM, Actor.USER],
        ),
    }
    
    def __init__(self, repository: StateRepository):
        self._repo = repository
        self._lock = Lock()
        self._timeout_handlers: Dict[State, Callable] = {}
        
    def transition(
        self,
        topic: str,
        to_state: State,
        actor: Actor,
        reason: str = "",
        context: Dict = None
    ) -> 'TransitionResult':
        """
        执行状态转换
        
        Args:
            topic: 话题标识
            to_state: 目标状态
            actor: 执行者
            reason: 转换原因
            context: 附加上下文
            
        Returns:
            TransitionResult: 转换结果
        """
        with self._lock:
            current = self._repo.get_state(topic)
            current_state = current.state if current else None
            
            # 验证转换合法性
            validation = self._validate_transition(
                topic, current_state, to_state, actor
            )
            
            if not validation.is_valid:
                return TransitionResult(
                    success=False,
                    error=validation.error,
                    from_state=current_state,
                    to_state=to_state,
                )
            
            # 创建转换记录
            transition = StateTransition(
                from_state=current_state,
                to_state=to_state,
                actor=actor,
                reason=reason,
                context=context or {},
            )
            
            # 保存新状态
            new_state_info = StateInfo(
                topic=topic,
                state=to_state,
                history=current.history + [transition] if current else [transition],
                updated_at=datetime.now(),
            )
            
            self._repo.save_state(new_state_info)
            
            # 设置超时处理
            config = self.STATE_CONFIG.get(to_state)
            if config and config.timeout_seconds:
                self._schedule_timeout(topic, to_state, config.timeout_seconds)
            
            return TransitionResult(
                success=True,
                from_state=current_state,
                to_state=to_state,
                transition=transition,
            )
    
    def _validate_transition(
        self,
        topic: str,
        from_state: Optional[State],
        to_state: State,
        actor: Actor
    ) -> 'ValidationResult':
        """验证状态转换的合法性"""
        
        # 检查目标状态是否存在
        if to_state not in self.STATE_CONFIG:
            return ValidationResult(
                is_valid=False,
                error=f"Invalid target state: {to_state}"
            )
        
        config = self.STATE_CONFIG[to_state]
        
        # 如果是初始状态转换（从无到有）
        if from_state is None:
            if to_state != State.PENDING:
                return ValidationResult(
                    is_valid=False,
                    error=f"Initial state must be PENDING, got {to_state}"
                )
            return ValidationResult(is_valid=True)
        
        # 检查是否允许该转换
        if from_state not in config.allowed_transitions:
            return ValidationResult(
                is_valid=False,
                error=f"Transition from {from_state} to {to_state} not allowed"
            )
        
        # 检查执行者权限
        if actor not in config.valid_actors:
            return ValidationResult(
                is_valid=False,
                error=f"Actor {actor} not allowed for transition to {to_state}"
            )
        
        return ValidationResult(is_valid=True)
    
    def get_state(self, topic: str) -> Optional['StateInfo']:
        """获取话题当前状态"""
        return self._repo.get_state(topic)
    
    def get_history(self, topic: str) -> List[StateTransition]:
        """获取话题状态历史"""
        state = self._repo.get_state(topic)
        return state.history if state else []
    
    def on_timeout(self, topic: str, callback: Callable):
        """注册状态超时回调"""
        # 实现超时处理逻辑
        pass
```

```python
    增强的队列项
    
    携带完整血缘信息和状态历史
    """
    id: str
    topic: str
    status: State
    state_history: List[StateTransition] = field(default_factory=list)

    score: float = 0.0
    depth: float = 5.0
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)
    metadata: Dict = field(default_factory=dict)
    
    def __post_init__(self):
        if not self.id:
            self.id = self._generate_id()
    
    def _generate_id(self) -> str:
        """生成唯一ID"""
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S_%f")
        return f"cq_{timestamp}_{self.topic[:20]}"
    
    def to_dict(self) -> dict:
        return {
            "id": self.id,
            "topic": self.topic,
            "status": self.status.value,
            "state_history": [
                {
                    "from": t.from_state.value if t.from_state else None,

                    "actor": t.actor.value,
                    "timestamp": t.timestamp.isoformat(),
```

#### 3.2.2 服务实现

```python
class QueueService:
    """
    队列服务
    
    统一管理层级：
    - 入队/出队
    - 状态管理（委托给 StateMachine）
    - 持久化
    """
    
    def __init__(
        self,
        state_machine: ExplorationStateMachine,
        repository: QueueRepository,
    ):
        self._sm = state_machine
        self._repo = repository
        self._lock = Lock()
    
    def enqueue(
        self,
        topic: str,
        lineage: Optional[LineageInfo] = None,
        score: float = 0.0,
        depth: float = 5.0,
        actor: Actor = Actor.SYSTEM,
        reason: str = "",
    ) -> QueueItem:
        """
        入队
        
        自动设置状态为 PENDING
        """
        # 创建队列项
        item = QueueItem(
            topic=topic,
            status=State.PENDING,
            lineage=lineage or LineageInfo(),
            score=score,
            depth=depth,
        )
        
        # 使用状态机设置初始状态
        result = self._sm.transition(
            topic=topic,
            to_state=State.PENDING,
            actor=actor,
            reason=reason,
            context={"item_id": item.id},
        )
        
        if not result.success:
            raise QueueOperationError(f"Failed to enqueue: {result.error}")
        
        # 保存到仓库
        self._repo.save(item)
        
        return item
    
    def claim_next(
        self,
        agent_id: str,
        filter_fn: Optional[Callable[[QueueItem], bool]] = None,
    ) -> Optional[QueueItem]:
        """
        原子claim下一个待处理项
        
        自动更新状态: PENDING -> CLAIMED
        
        Args:
            agent_id: 请求claim的agent标识
            filter_fn: 可选的过滤函数
            
        Returns:
            QueueItem: 被claim的项，如果没有则返回None
        """
        with self._lock:
            # 获取所有pending项
            pending_items = self._repo.find_by_status(State.PENDING)
            
            # 按分数排序
            pending_items.sort(key=lambda x: x.score, reverse=True)
            
            # 应用自定义过滤
            if filter_fn:
                pending_items = [i for i in pending_items if filter_fn(i)]
            
            for item in pending_items:
                # 尝试状态转换
                result = self._sm.transition(
                    topic=item.topic,
                    to_state=State.CLAIMED,
                    actor=Actor.SPIDER_AGENT,
                    reason=f"Claimed by {agent_id}",
                    context={"agent_id": agent_id},
                )
                
                if result.success:
                    # 更新项
                    item.status = State.CLAIMED
                    item.state_history = result.transition_history
                    item.updated_at = datetime.now()
                    item.metadata["claimed_by"] = agent_id
                    item.metadata["claimed_at"] = datetime.now().isoformat()
                    
                    self._repo.save(item)
                    return item
            
            return None
    
    def start_exploration(self, item_id: str, agent_id: str) -> bool:
        """
        标记开始探索
        
        状态: CLAIMED -> EXPLORING
        """
        item = self._repo.get_by_id(item_id)
        if not item:
            return False
        
        result = self._sm.transition(
            topic=item.topic,
            to_state=State.EXPLORING,
            actor=Actor.SPIDER_AGENT,
            reason=f"Start exploration by {agent_id}",
        )
        
        if result.success:
            item.status = State.EXPLORING
            item.updated_at = datetime.now()
            self._repo.save(item)
            return True
        
        return False
    
    def complete_exploration(
        self,
        item_id: str,
        result: 'ExplorationResult',
    ) -> bool:
        """
        标记探索完成
        
        状态: EXPLORING -> DONE
        """
        item = self._repo.get_by_id(item_id)
        if not item:
            return False
        
        transition_result = self._sm.transition(
            topic=item.topic,
            to_state=State.DONE,
            actor=Actor.SPIDER_AGENT,
            reason="Exploration completed",
            context={"result": result.to_dict()},
        )
        
        if transition_result.success:
            item.status = State.DONE
            item.updated_at = datetime.now()
            item.metadata["exploration_result"] = result.to_dict()
            self._repo.save(item)
            return True
        
        return False
    
    def fail_exploration(
        self,
        item_id: str,
        error: str,
        retryable: bool = True,
    ) -> bool:
        """
        标记探索失败
        
        根据重试策略决定状态：FAILED 或 PENDING
        """
        item = self._repo.get_by_id(item_id)
        if not item:
            return False
        
        # 检查重试次数
        retry_count = item.metadata.get("retry_count", 0)
        max_retries = 3  # 可配置
        
        if retryable and retry_count < max_retries:
            # 重试：回到 PENDING
            new_state = State.PENDING
            item.metadata["retry_count"] = retry_count + 1
        else:
            # 不重试：标记 FAILED
            new_state = State.FAILED
        
        result = self._sm.transition(
            topic=item.topic,
            to_state=new_state,
            actor=Actor.SYSTEM,
            reason=f"Exploration failed: {error}",
            context={"error": error, "retry_count": item.metadata.get("retry_count", 0)},
        )
        
        if result.success:
            item.status = new_state
            item.updated_at = datetime.now()
            self._repo.save(item)
            return True
        
        return False
    
    def get_pending_count(self) -> int:
        """获取待处理数量"""
        return len(self._repo.find_by_status(State.PENDING))
    
    def get_stats(self) -> Dict[str, int]:
        """获取队列统计信息"""
        return {
            "pending": len(self._repo.find_by_status(State.PENDING)),
            "claimed": len(self._repo.find_by_status(State.CLAIMED)),
            "exploring": len(self._repo.find_by_status(State.EXPLORING)),
            "done": len(self._repo.find_by_status(State.DONE)),
            "failed": len(self._repo.find_by_status(State.FAILED)),
            "total": self._repo.count_all(),
        }
```
